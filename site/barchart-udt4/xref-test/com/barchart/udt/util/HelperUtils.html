56  	public static String getProperty(String name) {
57  
58  		String value = System.getProperty(name);
59  
60  		if (value == null) {
61  			log.error("property '{}' not defined; terminating", name);
62  			System.exit(1);
63  		}
64  
65  		return value;
66  
67  	}

69  	private static final AtomicInteger portCounter = new AtomicInteger(12345);
70  
71  	public static InetSocketAddress getLocalSocketAddress() {
72  
73  		final InetSocketAddress address = new InetSocketAddress("localhost",
74  				portCounter.getAndIncrement());
75  
76  		log.info("\n\t### allocated address={} ###", address);
77  
78  		return address;

80  	}

82  	static int[] randomIntArray(int length, int range) {
83  		int[] array = new int[length];
84  		Random generator = new Random();
85  		// for each item in the list
86  		for (int i = 0; i < array.length; i++) {
87  			// create a new random number and populate the
88  			// current location in the list with it
89  			array[i] = generator.nextInt(range);
90  		}
91  		return array;
92  	}
