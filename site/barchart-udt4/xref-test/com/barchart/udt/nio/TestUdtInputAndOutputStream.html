118 	private int copy(final InputStream is, final OutputStream os)
119 			throws IOException {
120 		if (is == null) {
121 			throw new NullPointerException("null input stream.");
122 		}
123 		if (os == null) {
124 			throw new NullPointerException("null output stream.");
125 		}
126 		final byte[] buffer = new byte[DEFAULT_BUFFER_SIZE];
127 		int count = 0;
128 		int n = 0;
129 		while (-1 != (n = is.read(buffer))) {
130 			os.write(buffer, 0, n);
131 			count += n;
132 		}
133 		log.trace("Wrote " + count + " bytes.");
134 		return count;
135 	}

143 	private long copy(final InputStream in, final OutputStream out,
144 			final int originalByteCount, final ReadStrategy readStrategy)
145 			throws IOException {
146 		if (originalByteCount < 0) {
147 			throw new IllegalArgumentException("Invalid byte count: "
148 					+ originalByteCount);
149 		}
150 		final int bufSize;
151 		if (originalByteCount < DEFAULT_BUFFER_SIZE) {
152 			bufSize = originalByteCount;
153 		} else {
154 			bufSize = DEFAULT_BUFFER_SIZE;
155 		}
156 		final byte buffer[] = new byte[bufSize];
157 		int len = 0;
158 		long written = 0;
159 		long byteCount = originalByteCount;
160 		try {
161 			while (byteCount > 0) {
162 				log.info("IN LOOP");
163 				if (byteCount < bufSize) {
164 					len = readStrategy.read(in, buffer, 0, (int) byteCount);
165 					// len = in.read(buffer, 0, (int) byteCount);
166 				} else {
167 					len = readStrategy.read(in, buffer, 0, bufSize);
168 					// len = in.read(buffer, 0, bufSize);
169 				}
170 				if (len == -1) {
171 					log.info("Breaking on length = -1");
172 					break;
173 				}
174 				byteCount -= len;
175 				log.info("Decrementing byte count by " + len + " to "
176 						+ byteCount);
177 				out.write(buffer, 0, len);
178 				log.info("WROTE DATA");
179 				written += len;
180 			}
181 			return written;
182 		} catch (final IOException e) {
183 			log.debug("Got IOException during copy after writing " + written
184 					+ " of " + originalByteCount, e);
185 			e.printStackTrace();
186 			throw e;
187 		} catch (final RuntimeException e) {
188 			log.debug("Runtime error after writing " + written + " of "
189 					+ originalByteCount, e);
190 			e.printStackTrace();
191 			throw e;
192 		} finally {
193 			out.flush();
194 		}
195 	}
