113 	private void startUdtServer(final InetSocketAddress serverAddress,
114 			final AtomicReference<String> ref, 
115 			final AtomicBoolean readyToAccept) throws IOException {
116 
117 		final ServerSocket acceptorSocket = new NetServerSocketUDT();
118 
119 		acceptorSocket.bind(serverAddress);
120 		assert acceptorSocket.isBound();
121 
122 		readyToAccept.set(true);
123 		synchronized(readyToAccept) {
124 			readyToAccept.notifyAll();
125 		}
126 		final Socket connectorSocket = acceptorSocket.accept();
127 		assert connectorSocket.isConnected();
128 
129 		echo(connectorSocket, ref);
130 
131 	}
