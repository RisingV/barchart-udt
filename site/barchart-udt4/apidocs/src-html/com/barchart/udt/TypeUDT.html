040    package com.barchart.udt;
041    
042    /**
043     * UDT socket mode type.
044     * 
045     * NOTE: {@link #TypeUDT} means stream vs datagram;
046     * {@link com.barchart.udt.nio.Kind} means server vs client.
047     * <p>
048     * maps to socket.h constants<br>
049     * SOCK_STREAM = 1<br>
050     * SOCK_DGRAM = 2<br>
051     */
052    public enum TypeUDT {
053    
054            /** The STREAM type. Defines stream - oriented UDT mode. */
055            STREAM(1), //
056    
057            /** The DATAGRAM. Defines datagram/message - oriented UDT mode. */
058            DATAGRAM(2), //
059    
060            /** The code. */
061            ;
062    
063            public final int code;
064    
065            /**
066             * Instantiates a new type udt.
067             * 
068             * @param code
069             *            the code
070             */
071            private TypeUDT(int code) {
072                    this.code = code;
073            }
074    
075    }
