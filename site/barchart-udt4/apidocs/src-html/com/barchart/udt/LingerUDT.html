040    package com.barchart.udt;
041    
042    public class LingerUDT extends Number implements Comparable<LingerUDT> {
043    
044            public static final LingerUDT LINGER_ZERO = new LingerUDT(0);
045    
046            // measured in seconds
047            final int timeout;
048    
049            /**
050             * Default constructor. NOTE: linger value is "u_short" on windows and "int"
051             * on linux:<br>
052             * Windows: <a
053             * href="http://msdn.microsoft.com/en-us/library/ms739165(VS.85).aspx">
054             * linger Structure on Windows</a><br>
055             * Linux: <a href=
056             * "http://www.gnu.org/s/libc/manual/html_node/Socket_002dLevel-Options.html"
057             * >GCC Socket-Level Options</a><br>
058             * Therefore select smallest range: 0 <= linger <= 65535 <br>
059             * 
060             * @param lingerSeconds
061             *            the seconds to linger; "0" means "do not linger"
062             * 
063             * @throws IllegalArgumentException
064             *             when lingerSeconds is out of range
065             */
066            public LingerUDT(int lingerSeconds) throws IllegalArgumentException {
067                    if (65535 < lingerSeconds) {
068                            throw new IllegalArgumentException(
069                                            "lingerSeconds is out of range: 0 <= linger <= 65535");
070                    }
071                    this.timeout = lingerSeconds > 0 ? lingerSeconds : 0;
072            }
073    
074            private static final long serialVersionUID = 3414455799823407217L;
075    
076            @Override
077            public double doubleValue() {
078                    return timeout;
079            }
080    
081            /*
082             * (non-Javadoc)
083             * 
084             * @see java.lang.Number#floatValue()
085             */
086            @Override
087            public float floatValue() {
088                    return timeout;
089            }
090    
091            @Override
092            public int intValue() {
093                    return timeout;
094            }
095    
096            @Override
097            public long longValue() {
098                    return timeout;
099            }
100    
101            boolean isLingerOn() {
102                    return timeout > 0;
103            }
104    
105            int timeout() {
106                    return timeout;
107            }
108    
109            @Override
110            public boolean equals(Object otherLinger) {
111                    if (otherLinger instanceof LingerUDT) {
112                            LingerUDT other = (LingerUDT) otherLinger;
113                            return other.timeout == this.timeout;
114                    }
115                    return false;
116            }
117    
118            @Override
119            public int hashCode() {
120                    return timeout;
121            }
122    
123            @Override
124            public int compareTo(LingerUDT other) {
125                    return other.timeout - this.timeout;
126            }
127    
128            @Override
129            public String toString() {
130                    return String.valueOf(timeout);
131            }
132    
133    }
