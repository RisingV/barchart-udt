040    package com.barchart.udt;
041    
042    import java.net.SocketException;
043    
044    /**
045     * The Class ExceptionUDT. Wraps all native UDT exceptions and more.
046     */
047    // used by JNI; do not change any signatures;
048    public class ExceptionUDT extends SocketException {
049    
050            /** The Constant serialVersionUID. */
051            private static final long serialVersionUID = -6353519522691064012L;
052    
053            /**
054             * The error udt. Keeps error description for this exception. Use this enum
055             * in switch/case to fine tune exception processing.
056             */
057            public final ErrorUDT errorUDT;
058    
059            /**
060             * The socket id. Keeps socketID of the socket that produced this exception.
061             * Can possibly contain '0' when particular method can not determine
062             * {@link #socketID} that produced the exception.
063             */
064            public final int socketID;
065    
066            /**
067             * Instantiates a new exception udt for native UDT::Exception. This
068             * exception is generated in the underlying UDT method.
069             * 
070             * @param socketID
071             *            the socket id
072             * @param errorCode
073             *            the error code
074             * @param comment
075             *            the comment
076             */
077            protected ExceptionUDT(int socketID, int errorCode, String comment) {
078                    super(ErrorUDT.descriptionFrom(socketID, errorCode, comment));
079                    errorUDT = ErrorUDT.of(errorCode);
080                    this.socketID = socketID;
081            }
082    
083            // used by SocketUDT
084            /**
085             * Instantiates a new exception udt for synthetic JNI wrapper exception.
086             * This exception is generated in the JNI glue code itself.
087             * 
088             * @param socketID
089             *            the socket id
090             * @param error
091             *            the error
092             * @param comment
093             *            the comment
094             */
095            protected ExceptionUDT(int socketID, ErrorUDT error, String comment) {
096                    super(ErrorUDT.descriptionFrom(socketID, error.code, comment));
097                    errorUDT = error;
098                    this.socketID = socketID;
099            }
100    
101    }
