040    package com.barchart.udt.lib;
041    
042    import org.slf4j.Logger;
043    import org.slf4j.LoggerFactory;
044    
045    public enum LibraryUDT_3 {
046    
047            UNKNOWN(null, "xxx.xxx.xxx"), //
048    
049            I386_LINUX_GPP(null, "i386.Linux.g++"), //
050    
051            AMD64_LINUX_GPP(null, "amd64.Linux.g++"), //
052    
053            I386_MACOSX_GPP(null, "i386.MacOSX.g++"), //
054    
055            X86_64_MACOSX_GPP(null, "x86_64.MacOSX.g++"), //
056    
057            // X86_WINDOWS_MSVC("msvcr90.dll,msvcp90.dll", "x86.Windows.msvc"), //
058            X86_WINDOWS_GPP("libgcc_s_dw2-1.dll,libstdc++-6.dll", "x86.Windows.g++"), //
059    
060            // X86_64_WINDOWS_MSVC("msvcr90.dll,msvcp90.dll", "x86_64.Windows.msvc"), //
061            X86_64_WINDOWS_GPP(null, "x86_64.Windows.g++"), //
062    
063            ;
064    
065            private final static Logger log = LoggerFactory
066                            .getLogger(LibraryUDT_3.class);
067    
068            static final String DOT = ".";
069            static final String COMMA = ",";
070            static final String SLASH = "/";
071            static final String DASH = "-";
072            static final String LIB = "lib";
073            static final String JNI = "jni";
074            static final String DEP = "dep";
075    
076            /** The Constant DEFAULT_EXTRACT_FOLDER_NAME. */
077            public final static String DEFAULT_EXTRACT_FOLDER_NAME = DOT + SLASH + LIB;
078    
079            final static String DEFAULT_LIB_DEP = SLASH + LIB + SLASH + DEP;
080    
081            private final String[] depsList;
082    
083            private final AOL aol;
084    
085            LibraryUDT_3(final String depNames, final String aolKey) {
086    
087                    this.depsList = getDeps(depNames);
088    
089                    this.aol = new AOL(aolKey);
090    
091            }
092    
093            String[] getDeps(final String depsNames) {
094                    if (depsNames == null) {
095                            return null;
096                    }
097                    final String[] depsList = depsNames.split(COMMA);
098                    if (depsList == null || depsList.length == 0) {
099                            return null;
100                    }
101                    return depsList;
102            }
103    
104            static LibraryUDT_3 detectLibrary() {
105                    for (final LibraryUDT_3 lib : values()) {
106                            if (lib.aol.isMatchJVM()) {
107                                    return lib;
108                            }
109                    }
110                    return UNKNOWN;
111            }
112    
113            String sourceDepPath(final String depName) {
114                    return DEFAULT_LIB_DEP + SLASH + aol.resourceName() + SLASH + depName;
115            }
116    
117            public static void load() throws Exception {
118                    load(null);
119            }
120    
121            public static void load(/* non-final */String targetFolder)
122                            throws Exception {
123    
124                    if (targetFolder == null || targetFolder.length() == 0) {
125                            targetFolder = DEFAULT_EXTRACT_FOLDER_NAME;
126                            log.warn("using default targetFolder={}", targetFolder);
127                    }
128    
129                    final LibraryUDT_3 library = detectLibrary();
130    
131                    //
132    
133                    library.loadDeps(targetFolder);
134    
135                    //
136    
137                    final String targetPath = library.targetPath(targetFolder);
138    
139                    try {
140                            log.info("source #1: CDT testing library");
141                            final String sourcePath = library.sourceTestCDT();
142                            RES.systemLoad(sourcePath, targetPath);
143                            return;
144                    } catch (Exception e) {
145                            log.warn("\n\t {} {}", e.getClass().getSimpleName(), e.getMessage());
146                    }
147    
148                    try {
149                            log.info("source #2: NAR testing library");
150                            final String sourcePath = library.sourceTestNAR();
151                            RES.systemLoad(sourcePath, targetPath);
152                            return;
153                    } catch (Exception e) {
154                            log.warn("\n\t {} {}", e.getClass().getSimpleName(), e.getMessage());
155                    }
156    
157                    try {
158                            log.info("source #3: NAR production library");
159                            final String sourcePath = library.sourceRealNAR();
160                            RES.systemLoad(sourcePath, targetPath);
161                            return;
162                    } catch (Exception e) {
163                            log.warn("\n\t {} {}", e.getClass().getSimpleName(), e.getMessage());
164                    }
165    
166                    throw new Exception("load failed");
167    
168            }
169    
170            void loadDeps(final String targetFoler) throws Exception {
171    
172                    if (depsList == null) {
173                            return;
174                    }
175    
176                    for (final String depName : depsList) {
177    
178                            log.info("\n\t depName={}", depName);
179    
180                            final String sourcePath = sourceDepPath(depName);
181    
182                            final String targetPath = targetDepPath(targetFoler, depName);
183    
184                            RES.systemLoad(sourcePath, targetPath);
185    
186                    }
187    
188            }
189    
190            String targetDepPath(final String targetFolder, final String depName) {
191                    return targetFolder + SLASH + aol.resourceName() + SLASH + depName;
192            }
193    
194            /**
195             * testing: custom CDT name convention; produced by CDT interactive build
196             * and moved to the target/test-classes/ to make it part of test classpath
197             */
198            // example:
199            // /libbarchart-i386-Linux-g++.so
200            String sourceTestCDT() {
201                    final String classifier = aol.resourceName();
202                    final String name = VersionUDT.BARCHART_ARTIFACT + DASH + classifier;
203                    final String library = System.mapLibraryName(name);
204                    final String path = //
205                    SLASH + library;
206                    return path;
207            }
208    
209            /**
210             * testing: maven-nar-plugin name convention; custom location:
211             * target/test-classes; part of java test classpath
212             */
213            // example:
214            // /libbarchart-i386-Linux-g++-jni/./lib/i386-Linux-g++/jni/libbarchart-1.0.2-SNAPSHOT.so
215            String sourceTestNAR() {
216                    final String classifier = aol.resourceName();
217                    final String name = VersionUDT.BARCHART_NAME;
218                    final String folder = name + DASH + classifier + DASH + JNI;
219                    final String path = //
220                    SLASH + folder + SLASH + sourceRealNAR();
221                    return path;
222            }
223    
224            /**
225             * production: maven-nar-plugin name convention; production location; part
226             * of production java classpath
227             */
228            // example:
229            // /lib/i386-Linux-g++/jni/libbarchart-1.0.2-SNAPSHOT.so
230            String sourceRealNAR() {
231                    final String name = VersionUDT.BARCHART_NAME;
232                    final String classifier = aol.resourceName();
233                    final String library = System.mapLibraryName(name);
234                    final String path = //
235                    SLASH + LIB + SLASH + classifier + SLASH + JNI + SLASH + library;
236                    return path;
237            }
238    
239            /**
240             * both testing and production: location and naming for extracted library;
241             * note that source artifact in jar (or on classpath) and the extracted
242             * artifact (on file system) will have different names
243             */
244            // example:
245            // ./lib/libbarchart-1.0.2-SNAPSHOT-i386-Linux-g++.so
246            String targetPath(final String targetFolder) {
247                    final String classifier = aol.resourceName();
248                    final String name = VersionUDT.BARCHART_NAME + DASH + classifier;
249                    final String library = System.mapLibraryName(name);
250                    final String path = //
251                    targetFolder + SLASH + library;
252                    return path;
253            }
