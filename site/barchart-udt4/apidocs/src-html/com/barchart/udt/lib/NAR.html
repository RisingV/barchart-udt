001    package com.barchart.udt.lib;
002    
003    import java.io.BufferedReader;
004    import java.io.InputStream;
005    import java.io.InputStreamReader;
006    import java.util.HashSet;
007    import java.util.Set;
008    
009    import org.slf4j.Logger;
010    import org.slf4j.LoggerFactory;
011    
012    public class NAR {
013    
014            /** The log. */
015            private final static Logger log = LoggerFactory.getLogger(NAR.class);
016    
017            static String readFileAsString(String filePath) throws Exception {
018    
019                    StringBuffer fileData = new StringBuffer(1024 * 4);
020    
021                    InputStream stream = NAR.class.getResourceAsStream(filePath);
022    
023                    BufferedReader reader = new BufferedReader(
024                                    new InputStreamReader(stream));
025    
026                    char[] buf = new char[1024];
027    
028                    int numRead = 0;
029    
030                    while ((numRead = reader.read(buf)) != -1) {
031                            String readData = String.valueOf(buf, 0, numRead);
032                            fileData.append(readData);
033                            buf = new char[1024];
034                    }
035    
036                    reader.close();
037    
038                    return fileData.toString();
039    
040            }
041    
042            enum SupportedLinker {
043    
044                    UNKNOWN(""), //
045    
046                    GPP("g++"), //
047    
048                    ;
049    
050                    final String name;
051    
052                    SupportedLinker(String name) {
053                            this.name = name;
054                    }
055    
056                    static SupportedLinker fromName(String name) {
057                            for (SupportedLinker linker : values()) {
058                                    if (linker.name.equals(name)) {
059                                            return linker;
060                                    }
061                            }
062                            return UNKNOWN;
063                    }
064    
065                    boolean isKnown() {
066                            return this != UNKNOWN;
067                    }
068    
069            }
070    
071            static final String NAR_AOL = "/nar-aol.properties";
072    
073            public static void main(String... args) throws Exception {
074    
075                    log.info("started");
076    
077                    String filePath = NAR_AOL;
078    
079                    String narAol = readFileAsString(filePath);
080    
081                    log.info("\n{}", narAol);
082    
083                    String[] lines = narAol.split("\n");
084    
085                    log.info("lines.length={}", lines.length);
086    
087                    Set<String> set = new HashSet<String>();
088    
089                    for (String line : lines) {
090    
091                            line = line.trim();
092    
093                            if (line.startsWith("#") || line.length() < 1) {
094                                    continue;
095                            }
096    
097                            AOL aol = new AOL(line);
098    
099                            if (aol.linker.contains("linker")) {
100                                    continue;
101                            }
102    
103                            if (!SupportedLinker.fromName(aol.linker).isKnown()) {
104                                    continue;
105                            }
106    
107                            String name = aol.resourceName();
108    
109                            set.add(name);
110    
111                    }
112    
113                    for (String line : set.toArray(new String[] {})) {
114    
115                            String find = AOL.filterArch() + "-" + AOL.filterName();
116    
117                            if (line.contains(find)) {
118                                    log.info("{}", line);
119                            }
120    
121                    }
122    
123            }
124    
125    }
