040    package com.barchart.udt.lib;
041    
042    import java.io.InputStream;
043    import java.util.Properties;
044    
045    import org.slf4j.Logger;
046    import org.slf4j.LoggerFactory;
047    
048    public class VersionUDT {
049    
050            static final Logger log = LoggerFactory.getLogger(VersionUDT.class);
051    
052            //
053    
054            static final String PROP_JAVA_VENDOR = "java.vendor";
055            static final String PROP_JAVA_VERSION = "java.version";
056            static final String PROP_JAVA_VM_NAME = "java.vm.name";
057    
058            public static final String JAVA_VENDOR = System
059                            .getProperty(PROP_JAVA_VENDOR);
060            public static final String JAVA_VERSION = System
061                            .getProperty(PROP_JAVA_VERSION);
062            public static final String JAVA_VM_NAME = System
063                            .getProperty(PROP_JAVA_VM_NAME);
064    
065            //
066    
067            static final String PROP_OS_NAME = "os.name";
068            static final String PROP_OS_ARCH = "os.arch";
069            static final String PROP_OS_VERSION = "os.version";
070    
071            public static final String OS_NAME = System.getProperty(PROP_OS_NAME);
072            public static final String OS_ARCH = System.getProperty(PROP_OS_ARCH);
073            public static final String OS_VERSION = System.getProperty(PROP_OS_VERSION);
074    
075            //
076    
077            static final String PROP_FILE = "version.properties";
078    
079            static final String PROP_UDT_VERSION = "udt.version";
080    
081            static final String PROP_BARCHART_NAME = "barchart.name";
082            static final String PROP_BARCHART_GROUP = "barchart.artifact";
083            static final String PROP_BARCHART_ARTIFACT = "barchart.artifact";
084            static final String PROP_BARCHART_VERSION = "barchart.version";
085            static final String PROP_BARCHART_TIMESTAMP = "barchart.timestamp";
086    
087            public static final String UDT_VERSION;
088            public static final String BARCHART_NAME;
089            public static final String BARCHART_GROUP;
090            public static final String BARCHART_ARTIFACT;
091            public static final String BARCHART_VERSION;
092            public static final String BARCHART_TIMESTAMP;
093    
094            static final String UNKNOWN = "UNKNOWN";
095    
096            static {
097    
098                    String udtVersion = UNKNOWN;
099                    String name = UNKNOWN;
100                    String group = UNKNOWN;
101                    String artifact = UNKNOWN;
102                    String version = UNKNOWN;
103                    String timestamp = UNKNOWN;
104    
105                    try {
106    
107                            final Properties props = new Properties();
108    
109                            final InputStream stream = VersionUDT.class.getClassLoader()
110                                            .getResourceAsStream(PROP_FILE);
111    
112                            props.load(stream);
113    
114                            udtVersion = props.getProperty(PROP_UDT_VERSION);
115    
116                            name = props.getProperty(PROP_BARCHART_NAME);
117                            group = props.getProperty(PROP_BARCHART_GROUP);
118                            artifact = props.getProperty(PROP_BARCHART_ARTIFACT);
119                            version = props.getProperty(PROP_BARCHART_VERSION);
120                            timestamp = props.getProperty(PROP_BARCHART_TIMESTAMP);
121    
122                    } catch (Exception e) {
123                            log.error("failed to load version properties", e);
124                    }
125    
126                    UDT_VERSION = udtVersion;
127    
128                    BARCHART_NAME = name;
129                    BARCHART_GROUP = group;
130                    BARCHART_ARTIFACT = artifact;
131                    BARCHART_VERSION = version;
132                    BARCHART_TIMESTAMP = timestamp;
133    
134            }
135    
136            private static void append(StringBuilder text, String EOL) {
137    
138                    text.append(PROP_BARCHART_ARTIFACT);
139                    text.append(" = ");
140                    text.append(BARCHART_ARTIFACT);
141                    text.append(EOL);
142    
143                    text.append(PROP_BARCHART_VERSION);
144                    text.append(" = ");
145                    text.append(BARCHART_VERSION);
146                    text.append(EOL);
147    
148                    text.append(PROP_BARCHART_TIMESTAMP);
149                    text.append(" = ");
150                    text.append(BARCHART_TIMESTAMP);
151                    text.append(EOL);
152    
153                    text.append(PROP_JAVA_VENDOR);
154                    text.append(" = ");
155                    text.append(JAVA_VENDOR);
156                    text.append(EOL);
157    
158                    text.append(PROP_JAVA_VERSION);
159                    text.append(" = ");
160                    text.append(JAVA_VERSION);
161                    text.append(EOL);
162    
163                    text.append(PROP_JAVA_VM_NAME);
164                    text.append(" = ");
165                    text.append(JAVA_VM_NAME);
166                    text.append(EOL);
167    
168                    text.append(PROP_OS_NAME);
169                    text.append(" = ");
170                    text.append(OS_NAME);
171                    text.append(EOL);
172    
173                    text.append(PROP_OS_ARCH);
174                    text.append(" = ");
175                    text.append(OS_ARCH);
176                    text.append(EOL);
177    
178                    text.append(PROP_OS_VERSION);
179                    text.append(" = ");
180                    text.append(OS_VERSION);
181                    text.append(EOL);
182    
183            }
184    
185            public static final void log() {
186                    log.info("\n{}", asText());
187            }
188    
189            public static final String asText() {
190    
191                    final StringBuilder text = new StringBuilder(128);
192    
193                    text.append("\n");
194                    text.append("#######################################");
195                    text.append("\n");
196    
197                    append(text, "\n");
198    
199                    text.append("#######################################");
200                    text.append("\n");
201                    text.append("\n");
202    
203                    return text.toString();
204    
205            }
206    
207            public static final String asHtml() {
208    
209                    final StringBuilder text = new StringBuilder(128);
210    
211                    text.append("<html><pre>");
212    
213                    append(text, "<br>");
214    
215                    text.append("</pre></html>");
216    
217                    return text.toString();
218    
219            }
220    
221    }
