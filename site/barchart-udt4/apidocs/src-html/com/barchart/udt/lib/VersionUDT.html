001    package com.barchart.udt.lib;
002    
003    import java.io.InputStream;
004    import java.util.Properties;
005    
006    import org.slf4j.Logger;
007    import org.slf4j.LoggerFactory;
008    
009    public class VersionUDT {
010    
011            static final Logger log = LoggerFactory.getLogger(VersionUDT.class);
012    
013            //
014    
015            static final String PROP_JAVA_VENDOR = "java.vendor";
016            static final String PROP_JAVA_VERSION = "java.version";
017            static final String PROP_JAVA_VM_NAME = "java.vm.name";
018    
019            public static final String JAVA_VENDOR = System
020                            .getProperty(PROP_JAVA_VENDOR);
021            public static final String JAVA_VERSION = System
022                            .getProperty(PROP_JAVA_VERSION);
023            public static final String JAVA_VM_NAME = System
024                            .getProperty(PROP_JAVA_VM_NAME);
025    
026            //
027    
028            static final String PROP_OS_NAME = "os.name";
029            static final String PROP_OS_ARCH = "os.arch";
030            static final String PROP_OS_VERSION = "os.version";
031    
032            public static final String OS_NAME = System.getProperty(PROP_OS_NAME);
033            public static final String OS_ARCH = System.getProperty(PROP_OS_ARCH);
034            public static final String OS_VERSION = System.getProperty(PROP_OS_VERSION);
035    
036            //
037    
038            static final String PROP_FILE = "version.properties";
039    
040            static final String PROP_UDT_VERSION = "udt.version";
041    
042            static final String PROP_BARCHART_NAME = "barchart.name";
043            static final String PROP_BARCHART_GROUP = "barchart.artifact";
044            static final String PROP_BARCHART_ARTIFACT = "barchart.artifact";
045            static final String PROP_BARCHART_VERSION = "barchart.version";
046            static final String PROP_BARCHART_TIMESTAMP = "barchart.timestamp";
047    
048            public static final String UDT_VERSION;
049            public static final String BARCHART_NAME;
050            public static final String BARCHART_GROUP;
051            public static final String BARCHART_ARTIFACT;
052            public static final String BARCHART_VERSION;
053            public static final String BARCHART_TIMESTAMP;
054    
055            static final String UNKNOWN = "UNKNOWN";
056    
057            static {
058    
059                    String udtVersion = UNKNOWN;
060                    String name = UNKNOWN;
061                    String group = UNKNOWN;
062                    String artifact = UNKNOWN;
063                    String version = UNKNOWN;
064                    String timestamp = UNKNOWN;
065    
066                    try {
067    
068                            final Properties props = new Properties();
069    
070                            final InputStream stream = VersionUDT.class.getClassLoader()
071                                            .getResourceAsStream(PROP_FILE);
072    
073                            props.load(stream);
074    
075                            udtVersion = props.getProperty(PROP_UDT_VERSION);
076    
077                            name = props.getProperty(PROP_BARCHART_NAME);
078                            group = props.getProperty(PROP_BARCHART_GROUP);
079                            artifact = props.getProperty(PROP_BARCHART_ARTIFACT);
080                            version = props.getProperty(PROP_BARCHART_VERSION);
081                            timestamp = props.getProperty(PROP_BARCHART_TIMESTAMP);
082    
083                    } catch (Exception e) {
084                            log.error("failed to load version properties", e);
085                    }
086    
087                    UDT_VERSION = udtVersion;
088    
089                    BARCHART_NAME = name;
090                    BARCHART_GROUP = group;
091                    BARCHART_ARTIFACT = artifact;
092                    BARCHART_VERSION = version;
093                    BARCHART_TIMESTAMP = timestamp;
094    
095            }
096    
097            private static void append(StringBuilder text, String EOL) {
098    
099                    text.append(PROP_BARCHART_ARTIFACT);
100                    text.append(" = ");
101                    text.append(BARCHART_ARTIFACT);
102                    text.append(EOL);
103    
104                    text.append(PROP_BARCHART_VERSION);
105                    text.append(" = ");
106                    text.append(BARCHART_VERSION);
107                    text.append(EOL);
108    
109                    text.append(PROP_BARCHART_TIMESTAMP);
110                    text.append(" = ");
111                    text.append(BARCHART_TIMESTAMP);
112                    text.append(EOL);
113    
114                    text.append(PROP_JAVA_VENDOR);
115                    text.append(" = ");
116                    text.append(JAVA_VENDOR);
117                    text.append(EOL);
118    
119                    text.append(PROP_JAVA_VERSION);
120                    text.append(" = ");
121                    text.append(JAVA_VERSION);
122                    text.append(EOL);
123    
124                    text.append(PROP_JAVA_VM_NAME);
125                    text.append(" = ");
126                    text.append(JAVA_VM_NAME);
127                    text.append(EOL);
128    
129                    text.append(PROP_OS_NAME);
130                    text.append(" = ");
131                    text.append(OS_NAME);
132                    text.append(EOL);
133    
134                    text.append(PROP_OS_ARCH);
135                    text.append(" = ");
136                    text.append(OS_ARCH);
137                    text.append(EOL);
138    
139                    text.append(PROP_OS_VERSION);
140                    text.append(" = ");
141                    text.append(OS_VERSION);
142                    text.append(EOL);
143    
144            }
145    
146            public static final void log() {
147                    log.info("\n{}", asText());
148            }
149    
150            public static final String asText() {
151    
152                    final StringBuilder text = new StringBuilder(128);
153    
154                    text.append("\n");
155                    text.append("#######################################");
156                    text.append("\n");
157    
158                    append(text, "\n");
159    
160                    text.append("#######################################");
161                    text.append("\n");
162                    text.append("\n");
163    
164                    return text.toString();
165    
166            }
167    
168            public static final String asHtml() {
169    
170                    final StringBuilder text = new StringBuilder(128);
171    
172                    text.append("<html><pre>");
173    
174                    append(text, "<br>");
175    
176                    text.append("</pre></html>");
177    
178                    return text.toString();
179    
180            }
181    
182    }
