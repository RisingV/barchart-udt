040    package com.barchart.udt.net;
041    
042    import java.io.IOException;
043    import java.io.InputStream;
044    import java.io.OutputStream;
045    import java.net.InetAddress;
046    import java.net.InetSocketAddress;
047    import java.net.Socket;
048    import java.net.SocketAddress;
049    import java.net.SocketException;
050    import java.nio.channels.SocketChannel;
051    
052    import com.barchart.udt.ExceptionUDT;
053    import com.barchart.udt.SocketUDT;
054    import com.barchart.udt.TypeUDT;
055    
056    public class NetSocketUDT extends Socket implements IceSocket {
057    
058            protected InputStream inputStream;
059            protected OutputStream outputStream;
060    
061            protected final SocketUDT socketUDT;
062    
063            public NetSocketUDT() throws ExceptionUDT {
064                    this.socketUDT = new SocketUDT(TypeUDT.STREAM);
065                    this.socketUDT.configureBlocking(true);
066            }
067    
068            /** NOTE: you just carefully choose TypeUDT */
069            public NetSocketUDT(final SocketUDT socketUDT) {
070                    this.socketUDT = socketUDT;
071            }
072    
073            @Override
074            public void bind(SocketAddress bindpoint) throws IOException {
075                    if (bindpoint == null) {
076                            bindpoint = new InetSocketAddress(0);
077                    }
078                    socketUDT.bind((InetSocketAddress) bindpoint);
079            }
080    
081            @Override
082            public void close() throws IOException {
083                    socketUDT.close();
084            }
085    
086            @Override
087            public void connect(SocketAddress endpoint) throws IOException {
088                    socketUDT.connect((InetSocketAddress) endpoint);
089            }
090    
091            @Override
092            public void connect(SocketAddress endpoint, int timeout) throws IOException {
093                    throw new RuntimeException("feature not available");
094            }
095    
096            @Override
097            public SocketChannel getChannel() {
098                    throw new RuntimeException("feature not available");
099            }
100    
101            @Override
102            public InetAddress getInetAddress() {
103                    return socketUDT.getRemoteInetAddress();
104            }
105    
106            @Override
107            public synchronized InputStream getInputStream() throws IOException {
108                    if (inputStream == null) {
109                            inputStream = new NetInputStreamUDT(socketUDT);
110                    }
111                    return inputStream;
112            }
113    
114            @Override
115            public boolean getKeepAlive() throws SocketException {
116                    throw new RuntimeException("feature not available");
117            }
118    
119            @Override
120            public InetAddress getLocalAddress() {
121                    return socketUDT.getLocalInetAddress();
122            }
123    
124            @Override
125            public int getLocalPort() {
126                    return socketUDT.getLocalInetPort();
127            }
128    
129            @Override
130            public SocketAddress getLocalSocketAddress() {
131                    try {
132                            return socketUDT.getLocalSocketAddress();
133                    } catch (ExceptionUDT e) {
134                            return null;
135                    }
136            }
137    
138            @Override
139            public boolean getOOBInline() throws SocketException {
140                    throw new RuntimeException("feature not available");
141            }
142    
143            @Override
144            public synchronized OutputStream getOutputStream() throws IOException {
145                    if (outputStream == null) {
146                            outputStream = new NetOutputStreamUDT(socketUDT);
147                    }
148                    return outputStream;
149            }
150    
151            @Override
152            public int getPort() {
153                    return socketUDT.getRemoteInetPort();
154            }
155    
156            @Override
157            public int getReceiveBufferSize() throws SocketException {
158                    return socketUDT.getReceiveBufferSize();
159            }
160    
161            @Override
162            public SocketAddress getRemoteSocketAddress() {
163                    try {
164                            return socketUDT.getRemoteSocketAddress();
165                    } catch (ExceptionUDT e) {
166                            return null;
167                    }
168            }
169    
170            @Override
171            public boolean getReuseAddress() throws SocketException {
172                    return socketUDT.getReuseAddress();
173            }
174    
175            @Override
176            public int getSendBufferSize() throws SocketException {
177                    return socketUDT.getSendBufferSize();
178            }
179    
180            @Override
181            public int getSoLinger() throws SocketException {
182                    return socketUDT.getSoLinger();
183            }
184    
185            @Override
186            public int getSoTimeout() throws SocketException {
187                    return socketUDT.getSoTimeout();
188            }
189    
190            @Override
191            public boolean getTcpNoDelay() throws SocketException {
192                    throw new RuntimeException("feature not available");
193            }
194    
195            @Override
196            public int getTrafficClass() throws SocketException {
197                    throw new RuntimeException("feature not available");
198            }
199    
200            @Override
201            public boolean isBound() {
202                    return socketUDT.isBound();
203            }
204    
205            @Override
206            public boolean isClosed() {
207                    return socketUDT.isClosed();
208            }
209    
210            @Override
211            public boolean isConnected() {
212                    return socketUDT.isConnected();
213            }
214    
215            @Override
216            public boolean isInputShutdown() {
217                    return socketUDT.isClosed();
218            }
219    
220            @Override
221            public boolean isOutputShutdown() {
222                    return socketUDT.isClosed();
223            }
224    
225            @Override
226            public void sendUrgentData(int data) throws IOException {
227                    throw new RuntimeException("feature not available");
228            }
229    
230            @Override
231            public void setKeepAlive(boolean on) throws SocketException {
232                    throw new RuntimeException("feature not available");
233            }
234    
235            @Override
236            public void setOOBInline(boolean on) throws SocketException {
237                    throw new RuntimeException("feature not available");
238            }
239    
240            @Override
241            public void setPerformancePreferences(int connectionTime, int latency,
242                            int bandwidth) {
243                    throw new RuntimeException("feature not available");
244            }
245    
246            @Override
247            public void setReceiveBufferSize(int size) throws SocketException {
248                    socketUDT.setReceiveBufferSize(size);
249            }
250    
251            @Override
252            public void setReuseAddress(boolean on) throws SocketException {
253                    socketUDT.setReuseAddress(on);
254            }
255    
256            @Override
257            public void setSendBufferSize(int size) throws SocketException {
258                    socketUDT.setSendBufferSize(size);
259            }
260    
261            @Override
262            public void setSoLinger(boolean on, int linger) throws SocketException {
263                    socketUDT.setSoLinger(on, linger);
264            }
265    
266            @Override
267            public void setSoTimeout(int timeout) throws SocketException {
268                    socketUDT.setSoTimeout(timeout);
269            }
270    
271            @Override
272            public void setTcpNoDelay(boolean on) throws SocketException {
273                    throw new RuntimeException("feature not available");
274            }
275    
276            @Override
277            public void setTrafficClass(int tc) throws SocketException {
278                    throw new RuntimeException("feature not available");
279            }
280    
281            @Override
282            public void shutdownInput() throws IOException {
283                    socketUDT.close();
284            }
285    
286            @Override
287            public void shutdownOutput() throws IOException {
288                    socketUDT.close();
289            }
290    
291    }
