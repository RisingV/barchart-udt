040    package com.barchart.udt.nio;
041    
042    import java.io.IOException;
043    import java.nio.channels.DatagramChannel;
044    import java.nio.channels.Pipe;
045    import java.nio.channels.ServerSocketChannel;
046    import java.nio.channels.SocketChannel;
047    import java.nio.channels.spi.AbstractSelector;
048    import java.nio.channels.spi.SelectorProvider;
049    
050    import com.barchart.udt.SocketUDT;
051    import com.barchart.udt.TypeUDT;
052    
053    /**
054     * you must use the same system-wide provider instance for the same
055     * {@link TypeUDT} of UDT channels and UDT selectors;
056     */
057    public class SelectorProviderUDT extends SelectorProvider {
058    
059            /**
060             * system-wide provider instance, for {@link TypeUDT#DATAGRAM} UDT sockets
061             */
062            public static final SelectorProviderUDT DATAGRAM = //
063            new SelectorProviderUDT(TypeUDT.DATAGRAM);
064    
065            /**
066             * system-wide provider instance, for {@link TypeUDT#STREAM} UDT sockets
067             */
068            public static final SelectorProviderUDT STREAM = //
069            new SelectorProviderUDT(TypeUDT.STREAM);
070    
071            /**
072             * {@link TypeUDT} of UDT sockets generated by this provider
073             */
074            public final TypeUDT type;
075    
076            protected SelectorProviderUDT(TypeUDT type) {
077                    this.type = type;
078            }
079    
080            protected volatile int acceptQueueSize = SocketUDT.DEFAULT_ACCEPT_QUEUE_SIZE;
081    
082            public int getAcceptQueueSize() {
083                    return acceptQueueSize;
084            }
085    
086            public void setAcceptQueueSize(int queueSize) {
087                    acceptQueueSize = queueSize;
088            }
089    
090            protected volatile int maxSelectorSize = SocketUDT.DEFAULT_MAX_SELECTOR_SIZE;
091    
092            public int getMaxSelectorSize() {
093                    return maxSelectorSize;
094            }
095    
096            public void setMaxSelectorSize(int selectorSize) {
097                    maxSelectorSize = selectorSize;
098            }
099    
100            protected volatile int maxConnectorSize = SocketUDT.DEFAULT_CONNECTOR_POOL_SIZE;
101    
102            public int getMaxConnectorSize() {
103                    return maxConnectorSize;
104            }
105    
106            public void setMaxConnectorSize(int connectorSize) {
107                    maxConnectorSize = connectorSize;
108            }
109    
110            //
111    
112            /**
113             * feature not available
114             */
115            @Override
116            public DatagramChannel openDatagramChannel() throws IOException {
117                    throw new RuntimeException("feature not available");
118            }
119    
120            /**
121             * feature not available
122             */
123            @Override
124            public Pipe openPipe() throws IOException {
125                    throw new RuntimeException("feature not available");
126            }
127    
128            @Override
129            public AbstractSelector openSelector() throws IOException {
130                    return new SelectorUDT(this, maxSelectorSize, maxConnectorSize);
131            }
132    
133            @Override
134            public ServerSocketChannel openServerSocketChannel() throws IOException {
135                    SocketUDT serverSocketUDT = new SocketUDT(type);
136                    return new ChannelServerSocketUDT(this, serverSocketUDT);
137            }
138    
139            @Override
140            public SocketChannel openSocketChannel() throws IOException {
141                    SocketUDT socketUDT = new SocketUDT(type);
142                    return new ChannelSocketUDT(this, socketUDT);
143            }
144    
145    }
