084            protected static Selector open(TypeUDT type) throws IOException {
085                    final SelectorProviderUDT provider;
086                    switch (type) {
087                    case DATAGRAM:
088                            provider = SelectorProviderUDT.DATAGRAM;
089                            break;
090                    case STREAM:
091                            provider = SelectorProviderUDT.STREAM;
092                            break;
093                    default:
094                            log.error("unsupported type={}", type);
095                            throw new IOException("unsupported type");
096                    }
097                    return provider.openSelector();
098            }

102            @Override
103            protected void implCloseSelector() throws IOException {
104                    wakeup();
105                    synchronized (this) {
106                            synchronized (publicRegisteredKeySet) {
107                                    synchronized (publicSelectedKeySet) {
108    
109                                            selectedKeySet.clear();
110    
111                                            for (final SelectionKey key : registeredKeySet) {
112                                                    try {
113                                                            key.channel().close();
114                                                    } catch (Throwable e) {
115                                                            log.error("unexpected", e);
116                                                    }
117                                            }
118    
119                                            registeredKeySet.clear();
120    
121                                    }
122                            }
123                    }
124            }

126            // NOE: register() and select() are blocking each other
127            @Override
128            protected SelectionKey register(AbstractSelectableChannel channel,
129                            int interestOps, Object attachment) {
130    
131                    if (!(channel instanceof ChannelUDT)) {
132                            // also takes care of null
133                            log.error("!(channel instanceof ChannelUDT)");
134                            throw new IllegalSelectorException();
135                    }
136    
137                    synchronized (publicRegisteredKeySet) {
138    
139                            if (publicRegisteredKeySet.size() == maximimSelectorSize) {
140                                    log.error("reached maximimSelectorSize)");
141                                    throw new IllegalSelectorException();
142                            }
143    
144                            ChannelUDT channelUDT = (ChannelUDT) channel;
145    
146                            SocketUDT socketUDT = channelUDT.getSocketUDT();
147    
148                            SelectionKeyUDT keyUDT = new SelectionKeyUDT(//
149                                            this, channelUDT, attachment, interestOps);
150    
151                            // XXX the only place with "add/put"
152                            registeredKeyMap.put(socketUDT.getSocketId(), keyUDT);
153                            registeredKeySet.add(keyUDT);
154    
155                            return keyUDT;
156    
157                    }
158    
159            }

184            @Override
185            // per java.nio.Selector contract 0 input means infinite
186            public int select(long timeout) throws IOException {
187                    if (timeout < 0) {
188                            throw new IllegalArgumentException("Negative timeout");
189                    }
190                    long timeoutUDT = (timeout == 0) ? UDT_TIMEOUT_INFINITE : timeout;
191                    return doSelectInsideLock(timeoutUDT);
192            }

194            @Override
195            public int selectNow() throws IOException {
196                    return doSelectInsideLock(UDT_TIMEOUT_NONE);
197            }

199            /**
200             * Wakeup the selector. NOTE: it is recommended to user timed
201             * {@link #select(long)} (say, timeout = 100 milliseconds) to avoid missed
202             * {@link #wakeup()} calls which are rare but possible in current
203             * implementation (accepted in favor of design simplicity / performance)
204             */
205            @Override
206            public Selector wakeup() {
207                    // publisher for volatile
208                    wakeupStepCount++;
209                    return this;
210            }

222                    return wakeupBaseCount != wakeupStepCount;

325            private final int doSelectInsideLock(long millisTimeout) throws IOException {
326                    if (!isOpen()) {
327                            throw new ClosedSelectorException();
328                    }
329                    synchronized (this) {
330                            synchronized (publicRegisteredKeySet) {
331                                    synchronized (publicSelectedKeySet) {
332                                            return doSelectReally(millisTimeout);
333                                    }
334                            }
335                    }
336            }

404                    if (isBufferBased) {

450            private final void prepareInterest() {
451    
452                    int readSize = 0;
453                    int writeSize = 0;
454                    int exceptSize = 0;
455    
456                    for (final SelectionKeyUDT keyUDT : registeredKeySet) {
457    
458                            final ChannelUDT channelUDT = keyUDT.channelUDT;
459                            keyUDT.readyOps = 0; // publisher for volatile
460    
461                            if (channelUDT.isOpenSocketUDT()) {
462    
463                                    final int interestOps = keyUDT.interestOps;
464                                    final int socketID = keyUDT.socketID;
465                                    final KindUDT channelType = channelUDT.getChannelKind();
466    
467                                    if ((interestOps & (OP_ACCEPT)) != 0) {
468                                            assert channelType == KindUDT.ACCEPTOR;
469                                            setReadInterest(readSize, socketID);
470                                            readSize++;
471                                    }
472                                    if ((interestOps & (OP_READ)) != 0) {
473                                            assert channelType == KindUDT.CONNECTOR;
474                                            setReadInterest(readSize, socketID);
475                                            readSize++;
476                                    }
477                                    if ((interestOps & (OP_WRITE)) != 0) {
478                                            assert channelType == KindUDT.CONNECTOR;
479                                            setWriteInterest(writeSize, socketID);
480                                            writeSize++;
481                                    }
482                                    if ((interestOps & (OP_CONNECT)) != 0) {
483                                            assert channelType == KindUDT.CONNECTOR;
484                                            assert (interestOps & (OP_READ | OP_WRITE | OP_ACCEPT)) == 0;
485                                            /*
486                                             * UDT does not support select() for connect() operation as
487                                             * yet; using thread pool instead
488                                             */
489                                    }
490                            } else {
491                                    synchronized (cancelledKeySet) {
492                                            cancelledKeySet.add(keyUDT);
493                                    }
494                            }
495                    }
496    
497                    // set sizes
498                    setInterestSize(readSize, writeSize, exceptSize);
499    
500            }
501    
502            private final void processInterest() {
503    
504                    // add ready to selected
505                    updateRead();
506                    updateWrite();
507                    updateExcept();
508    
509            }

511            /**
512             * note: millisTimeout input values contract:
513             * 
514             * -1 : infinite
515             * 
516             * =0 : immediate (but really up to 10 ms UDT resolution slice)
517             * 
518             * >0 : finite
519             */
520            private final int performSelect(long millisTimeout) throws ExceptionUDT {
521                    int readyCount = 0;
522                    saveWakeupBase();
523                    if (millisTimeout < 0) {
524                            /* infinite: do select in slices; check for wakeup; */
525                            do {
526                                    readyCount = doSelectSocketUDT(DEFAULT_MIN_SELECTOR_TIMEOUT);
527                                    if (readyCount > 0 || isWakeupPending()) {
528                                            break;
529                                    }
530                            } while (true);
531                    } else if (millisTimeout > 0) {
532                            /* finite: do select in slices; check for wakeup; count down */
533                            do {
534                                    readyCount = doSelectSocketUDT(DEFAULT_MIN_SELECTOR_TIMEOUT);
535                                    if (readyCount > 0 || isWakeupPending()) {
536                                            break;
537                                    }
538                                    millisTimeout -= DEFAULT_MIN_SELECTOR_TIMEOUT;
539                            } while (millisTimeout > 0);
540                    } else { // millisTimeout == 0
541                            /* immediate: one shot select */
542                            readyCount = doSelectSocketUDT(0);
543                    }
544                    return readyCount;
545            }

547            /**
548             * note: return value contract:
549             * 
550             * <0 : should not happen
551             * 
552             * =0 : means nothing was selected/timeout
553             * 
554             * >0 : number of pending r/w ops, NOT number of selected keys
555             */
556            private final int doSelectReally(long millisTimeout) throws IOException {
557    
558                    selectedKeySet.clear();
559    
560                    processCancelled();
561    
562                    int readyCount = 0;
563    
564                    trySelect: try {
565                            // java.nio.Selector contract for wakeup()
566                            begin();
567                            // pre select
568                            prepareInterest();
569                            // into select
570                            readyCount = performSelect(millisTimeout);
571                            if (readyCount == 0) {
572                                    // timeout, nothing is ready; no need to process post select
573                                    break trySelect;
574                            }
575                            // post select
576                            processInterest();
577                    } finally {
578                            // java.nio.Selector contract for wakeup()
579                            end();
580                    }
581    
582                    // using thread pool based connect() processor
583                    readyCount += updateConnect();
584    
585                    assert readyCount >= 0;
586    
587                    return readyCount;
588    
589            }
590    
591            private final int updateConnect() {
592                    final Queue<SelectionKeyUDT> readyQueue = connectorPool.readyQueue;
593                    if (readyQueue.isEmpty()) {
594                            return 0;
595                    }
596                    int updateCount = 0;
597                    SelectionKeyUDT keyUDT;
598                    while ((keyUDT = readyQueue.poll()) != null) {
599                            // contract:
600                            assert keyUDT.channelUDT.getChannelKind() == KindUDT.CONNECTOR;
601                            assert registeredKeySet.contains(keyUDT);
602                            assert (keyUDT.interestOps & OP_CONNECT) != 0;
603                            assert (keyUDT.interestOps & (OP_READ | OP_WRITE | OP_ACCEPT)) == 0;
604                            //
605                            keyUDT.readyOps |= OP_CONNECT;
606                            selectedKeySet.add(keyUDT);
607                            updateCount++;
608                    }
609                    return updateCount;
610            }

612            private final void updateRead() {
613                    final int readSize = getReadInterestSize();
614                    for (int index = 0; index < readSize; index++) {
615                            final int socketId = getReadInterest(index);
616                            final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
617                            assert keyUDT != null;
618                            switch (keyUDT.channelUDT.getChannelKind()) {
619                            case ACCEPTOR:
620                                    keyUDT.readyOps |= OP_ACCEPT;
621                                    break;
622                            case CONNECTOR:
623                                    keyUDT.readyOps |= OP_READ;
624                                    break;
625                            default:
626                                    assert false : "unexpected default";
627                                    continue;
628                            }
629                            selectedKeySet.add(keyUDT);
630                    }
631            }
632    
633            private final void updateWrite() {
634                    final int writeSize = getWriteInterestSize();
635                    for (int index = 0; index < writeSize; index++) {
636                            final int socketId = getWriteInterest(index);
637                            final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
638                            assert keyUDT != null;
639                            switch (keyUDT.channelUDT.getChannelKind()) {
640                            case ACCEPTOR:
641                                    assert false : "unexpected ACCEPTOR";
642                                    continue;
643                            case CONNECTOR:
644                                    keyUDT.readyOps |= OP_WRITE;
645                                    break;
646                            default:
647                                    assert false : "unexpected default";
648                                    continue;
649                            }
650                            selectedKeySet.add(keyUDT);
651                    }
652            }
653    
654            private final void updateExcept() {
655                    final int exceptSize = getExceptInterestSize();
656                    for (int k = 0; k < exceptSize; k++) {
657                            final int socketId = getExceptInterest(k);
658                            final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
659                            assert keyUDT != null;
660                            switch (keyUDT.channelUDT.getChannelKind()) {
661                            case ACCEPTOR:
662                            case CONNECTOR:
663                                    // set all ready OPS to throw on any operation
664                                    keyUDT.readyOps |= keyUDT.channel().validOps();
665                                    break;
666                            default:
667                                    assert false : "unexpected default";
668                                    continue;
669                            }
670                            selectedKeySet.add(keyUDT);
671                    }
672            }
673    
674            private final void processCancelled() throws IOException {
675                    /*
676                     * Precondition: Synchronized on this, publicRegisteredKeySet,
677                     * publicSelectedKeySet
678                     */
679                    synchronized (cancelledKeySet) {
680                            if (!cancelledKeySet.isEmpty()) {
681                                    for (final SelectionKey key : cancelledKeySet) {
682    
683                                            final SelectionKeyUDT keyUDT = (SelectionKeyUDT) key;
684    
685                                            // XXX the only place with "remove"
686    
687                                            final SelectionKeyUDT removed = //
688                                            registeredKeyMap.remove(keyUDT.socketID);
689                                            assert removed != null;
690    
691                                            final boolean isRemoved = //
692                                            registeredKeySet.remove(keyUDT);
693                                            assert isRemoved;
694    
695                                    }
696                                    cancelledKeySet.clear();
697                            }
698                    }
699            }

701            final void submitConnectRequest(SelectionKeyUDT keyUDT,
702                            InetSocketAddress remote) throws IOException {
703    
704                    // TODO think again if lack of sync is OK?
705    
706                    if (!registeredKeySet.contains(keyUDT)) {
707                            throw new IOException(//
708                                            "connect request while not registered");
709                    }
710    
711                    if ((keyUDT.interestOps & OP_CONNECT) == 0) {
712                            throw new IOException("connect request while not interested;"
713                                            + " key=" + keyUDT);
714                    }
715    
716                    if ((keyUDT.interestOps & (OP_ACCEPT | OP_READ | OP_WRITE)) != 0) {
717                            throw new IOException("connect request while is not sole interest;"
718                                            + " key=" + keyUDT);
719                    }
720    
721                    connectorPool.submitRequest(keyUDT, remote);
722    
723            }
