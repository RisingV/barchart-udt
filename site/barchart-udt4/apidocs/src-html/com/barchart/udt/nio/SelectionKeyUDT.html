040    package com.barchart.udt.nio;
041    
042    import java.net.InetSocketAddress;
043    import java.nio.channels.CancelledKeyException;
044    import java.nio.channels.SelectableChannel;
045    import java.nio.channels.SelectionKey;
046    import java.nio.channels.Selector;
047    import java.nio.channels.spi.AbstractSelectionKey;
048    
049    import com.barchart.udt.ExceptionUDT;
050    import com.barchart.udt.SocketUDT;
051    
052    /**
053     * NOTE: 1<->1 mapping between: SelectionKeyUDT -> ChannelUDT -> SocketUDT
054     */
055    public class SelectionKeyUDT extends AbstractSelectionKey {
056    
057            final SelectorUDT selectorUDT;
058            final ChannelUDT channelUDT;
059    
060            /**
061             * used by JNI
062             */
063            final int socketID;
064            final SocketUDT socketUDT;
065    
066            SelectionKeyUDT(SelectorUDT selector, ChannelUDT channel,
067                            Object attachment, int ops) {
068    
069                    this.channelUDT = channel;
070                    this.selectorUDT = selector;
071                    super.attach(attachment);
072                    this.interestOps = ops;
073    
074                    this.socketUDT = channel.getSocketUDT();
075                    this.socketID = socketUDT.getSocketId();
076    
077                    if (channel.getChannelKind() == KindUDT.CONNECTOR) {
078                            // special relationship for connectors
079                            ChannelSocketUDT socketChannel = (ChannelSocketUDT) channel;
080                            socketChannel.setRegistredKey(this);
081                    }
082    
083            }
084    
085            volatile int interestOps;
086            volatile int readyOps;
087    
088            //
089    
090            @Override
091            public SelectableChannel channel() {
092                    return (SelectableChannel) channelUDT;
093            }
094    
095            @Override
096            public int interestOps() {
097                    if (!isValid()) {
098                            throw new CancelledKeyException();
099                    }
100                    return interestOps;
101            }
102    
103            @Override
104            public SelectionKey interestOps(int ops) {
105                    if (!isValid()) {
106                            throw new CancelledKeyException();
107                    }
108                    if ((ops & ~(channel().validOps())) != 0) {
109                            throw new IllegalArgumentException("invalid ops");
110                    }
111                    if (((ops & OP_CONNECT) != 0) && socketUDT.isConnected()) {
112                            throw new IllegalArgumentException("already connected");
113                    }
114                    interestOps = ops;
115                    return this;
116            }
117    
118            @Override
119            public int readyOps() {
120                    if (!isValid()) {
121                            throw new CancelledKeyException();
122                    }
123                    return readyOps;
124            }
125    
126            @Override
127            public Selector selector() {
128                    return selectorUDT;
129            }
130    
131            //
132    
133            @Override
134            public int hashCode() {
135                    return socketID;
136            }
137    
138            @Override
139            public boolean equals(Object otherKey) {
140                    if (otherKey instanceof SelectionKeyUDT) {
141                            SelectionKeyUDT other = (SelectionKeyUDT) otherKey;
142                            return other.socketID == this.socketID;
143                    }
144                    return false;
145            }
146    
147            //
148    
149            @Override
150            public String toString() {
151                    InetSocketAddress local;
152                    try {
153                            local = socketUDT.getLocalSocketAddress();
154                    } catch (ExceptionUDT e) {
155                            local = null;
156                    }
157                    InetSocketAddress remote;
158                    try {
159                            remote = socketUDT.getRemoteSocketAddress();
160                    } catch (ExceptionUDT e) {
161                            remote = null;
162                    }
163    
164                    return " socketID=" + socketID
165                                    + //
166                                    " type=" + channelUDT.getChannelKind()
167                                    + //
168                                    " readyOps=" + toStringOps(readyOps) + "(" + readyOps + ")"
169                                    + //
170                                    " ntrstOps=" + toStringOps(interestOps) + "(" + interestOps
171                                    + ")" + //
172                                    " local=" + local + //
173                                    " remote=" + remote + //
174                                    "";
175            }
176    
177            public static final String toStringOps(int keyOps) {
178                    char R = (OP_READ & keyOps) != 0 ? 'R' : '-';
179                    char W = (OP_WRITE & keyOps) != 0 ? 'W' : '-';
180                    char C = (OP_CONNECT & keyOps) != 0 ? 'C' : '-';
181                    char A = (OP_ACCEPT & keyOps) != 0 ? 'A' : '-';
182                    return String.format("%c%c%c%c", A, C, W, R);
183            }
184    
185    }
