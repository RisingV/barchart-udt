040    package com.barchart.udt;
041    
042    /** status of underlying UDT native socket */
043    public enum StatusUDT {
044    
045            /* non UDT value */
046            UNKNOWN(0), //
047    
048            //
049    
050            /* keep in sync with udt.h UDTSTATUS enum */
051    
052            /** newly created socket; both connector and acceptor */
053            INIT(1), //
054    
055            /** just bound socket; both connector and acceptor */
056            OPENED(2), //
057    
058            /** bound + listening acceptor socket */
059            LISTENING(3), //
060    
061            /** connected connector socket */
062            CONNECTED(4), //
063    
064            /** acceptor socket after close() */
065            BROKEN(5), //
066    
067            /** connector socket after close() */
068            CLOSED(6), //
069    
070            /** trying to check status on socket that was closed and removed */
071            NONEXIST(7), //
072    
073            ;
074    
075            private final int code;
076    
077            private StatusUDT(final int code) {
078                    this.code = code;
079            }
080    
081            public int getCode() {
082                    return code;
083            }
084    
085            private static final StatusUDT[] ENUM_VALS = values();
086    
087            public static final StatusUDT fromCode(final int code) {
088                    for (final StatusUDT status : ENUM_VALS) {
089                            if (status.code == code) {
090                                    return status;
091                            }
092                    }
093                    return UNKNOWN;
094            }
095    
096            /**
097             * map UDT socket status to emulate JDK expected behavior
098             */
099            public boolean isOpenEmulateJDK() {
100                    switch (this) {
101                    case INIT:
102                    case OPENED:
103                    case LISTENING:
104                    case CONNECTED:
105                            return true;
106                    default:
107                            return false;
108                    }
109            }
110    
111    }
