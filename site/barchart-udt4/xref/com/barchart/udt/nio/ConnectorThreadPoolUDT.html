40  package com.barchart.udt.nio;

42  import java.net.InetSocketAddress;
43  import java.util.Queue;
44  import java.util.concurrent.ConcurrentHashMap;
45  import java.util.concurrent.ConcurrentLinkedQueue;
46  import java.util.concurrent.ConcurrentMap;
47  import java.util.concurrent.ExecutorService;
48  import java.util.concurrent.TimeUnit;
49  
50  class ConnectorThreadPoolUDT {
51  
52  	final ConnectorThreadFactoryUDT factory;
53  
54  	final Queue<SelectionKeyUDT> readyQueue = //
55  	new ConcurrentLinkedQueue<SelectionKeyUDT>();
56  
57  	final ConcurrentMap<SelectionKeyUDT, Runnable> taskMap = //
58  	new ConcurrentHashMap<SelectionKeyUDT, Runnable>();
59  
60  	final ExecutorService service;
61  
62  	final String THREAD_PREFIX = "UDT Connector #";
63  
64  	final int THREAD_PRIORITY = Thread.NORM_PRIORITY;
65  
66  	final int THREAD_TIME_KEEP = 3 * 1000;
67  	final TimeUnit THREAD_TIME_UNIT = TimeUnit.MILLISECONDS;
68  
69  	ConnectorThreadPoolUDT(int maximumPoolSize) {
70  
71  		factory = new ConnectorThreadFactoryUDT(THREAD_PREFIX, THREAD_PRIORITY);
72  
73  		// service = Executors.newCachedThreadPool();
74  
75  		service = ConnectorExecutorsUDT.newDynamicThreadPool(//
76  				maximumPoolSize, //
77  				THREAD_TIME_KEEP, //
78  				THREAD_TIME_UNIT, //
79  				factory);
80  
81  	}
82  
83  	void submitRequest(SelectionKeyUDT keyUDT, InetSocketAddress remote) {
84  
85  		if (taskMap.containsKey(keyUDT)) {
86  			return;
87  		}
88  
89  		Runnable task = new ConnectorTaskUDT(//
90  				keyUDT, taskMap, readyQueue, remote);
91  
92  		Runnable result = taskMap.putIfAbsent(keyUDT, task);
93  
94  		if (result == null) {
95  			service.submit(task);
96  		}
97  
98  	}

100 }
